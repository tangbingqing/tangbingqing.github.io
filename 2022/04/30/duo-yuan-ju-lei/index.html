层次聚类

from scipy.cluster.hierarchy import linkage,dendorgram,fcluster


# 执行分层/凝聚聚类

scipy.cluster.hierarchy.linkage(y, method='single', metric='euclidean', optimal_ordering=False)
### method : 计算形成的类之间距离
    single（最近点算法）、complete（最远点算法）、average（平均）、weighted（加权）、centroid（重心）、median（中位数）、ward（方差最小化算法）
### metric ：计算距离的方法
    ’euclidean’（欧几里得），’l1’，’l2’，’mantattan’（曼哈顿），’cosine’（余弦），’precomputed’（预先计算）

# ps : 如果linkage=’ward’，则metric必须为’euclidean’



# 从给定的链接矩阵定义的层次聚类中形成平面聚类，构造集群

fcluster(Z, t, criterion='inconsistent', depth=2, R=None, monocrit=None)
### Z ： ndarray ，linkage用函数返回的矩阵编码的层次聚类
### t ： 形成扁平集群时应用的阈值



# 层次聚类绘制为树状图

dendrogram (Z)

二、实验

1、选择不同的样本间举例或类间距离并观察聚类后的谱系图的变化情况

from scipy.cluster.hierarchy import dendrogram,linkage,fcluster
from sklearn.preprocessing import StandardScaler

data = pd.read_csv(r"WBClust2013.csv",header=0,index_col=0)

# 标准归一化
scal = StandardScaler()
data = scal.fit_transform(data)

# 聚类结果
metric = ['cityblock','euclidean','chebyshev','canberra','hamming','minkowski']
n = ["曼哈顿","欧几里得","切比雪夫","堪培拉","汉明距离","闵氏"]



plt.rcParams['font.sans-serif']=['SimHei']
plt.rcParams['axes.unicode_minus'] = False


for i in range (6):
    Z = linkage(data, method='single', metric = metric[i])
    fig = plt.figure(figsize=(5, 3))
    dn = dendrogram(Z)  # 制作谱系图
    plt.xlabel(n[i])
    plt.show()
